## 1. 概述

采用 **Room**（AndroidX SQLite ORM）实现，四个表对应 PRD 的完整实体定义。启用 WAL 模式 + 外键约束。

---

## 2. 实体关系

```
Source (1) ──→ (N) Resource (N) ──→ (M) Tag
                    │                       │
                    └── ResourceTag ───────┘  (junction)
```

## 4. 表定义

### 4.1 `sources` — 数据源

```kotlin
@Entity(tableName = "sources")
data class SourceEntity(
    @PrimaryKey val id: String,                    // UUID
    val name: String,                              // 用户自定义名称
    val type: SourceType,                          // LOCAL / SMB / FTP / WEBDAV
    val rootPath: String,                          // D:\Comics 或 smb://192.168.1.100/share
    val host: String? = null,                      // 网络源主机地址
    val port: Int? = null,                         // 默认 445
    val username: String? = null,
    val passwordStored: Boolean = false,            // 密码在 EncryptedSharedPreferences 中
    val domain: String? = null,                    // 域/工作组
    val enabled: Boolean = true,                   // 关闭后该源资源不在首页显示
    val isAvailable: Boolean = false,              // 当前是否可达
    val lastCheckAt: Long? = null,                 // 上次连接检查时间 (epoch ms)
    val createdAt: Long = System.currentTimeMillis(),
    val updatedAt: Long = System.currentTimeMillis(),
)
```

**密码安全**：
- `passwordStored = true` → 通过 `EncryptedSharedPreferences.getString("smb_pwd_$id")` 取密码
- `passwordStored = false` → 匿名访问
- Room 不存储任何密码数据

### 4.2 `resources` — 资源

```kotlin
@Entity(
    tableName = "resources",
    foreignKeys = [
        ForeignKey(
            entity = SourceEntity::class,
            parentColumns = ["id"],
            childColumns = ["sourceId"],
            onDelete = ForeignKey.CASCADE,
        )
    ],
    indices = [
        Index(value = ["sourceId"]),
        Index(value = ["sourceId", "relativePath"], unique = true),
        Index(value = ["createdAt", "id"]),
        Index(value = ["name", "id"]),
    ],
)
data class ResourceEntity(
    @PrimaryKey val id: String,                    // UUID
    val sourceId: String,                          // FK → sources.id
    val name: String,                              // 资源名称
    val type: ResourceType,                        // FOLDER / PDF / ARCHIVE / VIDEO
    val organizationMode: OrganizationMode? = null, // null = 未判定
    val relativePath: String,                      // 相对于源根目录的路径
    val thumbnailPath: String? = null,             // 本地缓存缩略图路径
    val fileCount: Int? = null,                    // 内部图片/页数
    val fileSize: Long? = null,                    // 字节
    val isAvailable: Boolean = true,               // 文件当前是否可访问
    val lastScannedAt: Long? = null,
    val createdAt: Long = System.currentTimeMillis(),
    val updatedAt: Long = System.currentTimeMillis(),
)
```

### 4.3 `tags` — 标签

```kotlin
@Entity(
```

### 4.4 `resource_tags` — 资源-标签关联

```kotlin
@Entity(
    tableName = "resource_tags",
    primaryKeys = ["resourceId", "tagId"],
    foreignKeys = [
        ForeignKey(
            entity = ResourceEntity::class,
            parentColumns = ["id"],
            childColumns = ["resourceId"],
            onDelete = ForeignKey.CASCADE,
        ),

            entity = TagEntity::class,
            parentColumns = ["id"],
            childColumns = ["tagId"],
            onDelete = ForeignKey.CASCADE,
        ),
    ],
    indices = [
        Index(value = ["resourceId", "tagId"]),
        Index(value = ["tagId", "resourceId"]),  // 标签筛选核心索引
    ],
)
data class ResourceTagEntity(
    val resourceId: String,
    val tagId: String,
    val createdAt: Long = System.currentTimeMillis(),
)
```

## 5. 级联删除

通过 Room `ForeignKey onDelete = CASCADE` + 外键约束保证：

| 操作 | Resource | Tag | ResourceTag |
|------|----------|-----|-------------|
| 删除 Source | 级联删除 | 保留 | 级联删除 |
| 删除 Resource | 删除 | 保留 | 级联删除 |
| 删除 Tag | 保留 | 删除 | 级联删除 |
| 资源不可达 | `isAvailable = false` | 不影响 | 不影响 |

## 6. DAO 定义

### 6.1 SourceDao

```kotlin
@Dao
interface SourceDao {
    @Query("SELECT * FROM sources ORDER BY createdAt DESC")
    fun getAllSources(): Flow<List<SourceEntity>>

    @Query("SELECT * FROM sources WHERE id = :id")
    suspend fun getSourceById(id: String): SourceEntity?

    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun insert(source: SourceEntity)

    @Update
    suspend fun update(source: SourceEntity)

    @Query("DELETE FROM sources WHERE id = :id")
    suspend fun deleteById(id: String)

    @Query("UPDATE sources SET isAvailable = :available, lastCheckAt = :checkTime WHERE id = :id")
    suspend fun updateAvailability(id: String, available: Boolean, checkTime: Long)
}
```

### 6.2 ResourceDao

```kotlin
@Dao
interface ResourceDao {
    @Query("SELECT * FROM resources WHERE isAvailable = 1 ORDER BY createdAt DESC")
    fun getAvailableResources(): Flow<List<ResourceEntity>>

    // 分页加载（Paging 3 集成，首页网格数据源。替代全量 Flow<List<>> 查询）
    @Query("""
        SELECT * FROM resources
        WHERE isAvailable = 1
        ORDER BY createdAt DESC
    """)
    fun getVisibleResourcesPagingSource(): PagingSource<Int, ResourceEntity>

    // 标签筛选分页
    @Query("""
        SELECT * FROM resources WHERE id IN (
            SELECT resourceId FROM resource_tags
            WHERE tagId IN (:tagIds)
            GROUP BY resourceId
            HAVING COUNT(DISTINCT tagId) = :tagCount
        )
        ORDER BY createdAt DESC
    """)
    fun filterByTagsPagingSource(tagIds: List<String>, tagCount: Int): PagingSource<Int, ResourceEntity>

    @Query("SELECT * FROM resources WHERE id = :id")
    suspend fun getById(id: String): ResourceEntity?

    @Insert(onConflict = OnConflictStrategy.IGNORE)
    suspend fun insert(resource: ResourceEntity)

    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun insertAll(resources: List<ResourceEntity>)

    @Update
    suspend fun update(resource: ResourceEntity)

    @Query("DELETE FROM resources WHERE id = :id")
    suspend fun deleteById(id: String)

    @Query("DELETE FROM resources WHERE sourceId = :sourceId")
    suspend fun deleteBySourceId(sourceId: String)

    // 按名称搜索
    @Query("SELECT * FROM resources WHERE name LIKE '%' || :query || '%' ORDER BY name ASC")
    fun searchByName(query: String): Flow<List<ResourceEntity>>
}
```

## 7. 内置标签播种

```kotlin
@Database(entities = [...], version = 1)
abstract class AppDatabase : RoomDatabase() {
    abstract fun sourceDao(): SourceDao
    abstract fun resourceDao(): ResourceDao
    abstract fun tagDao(): TagDao
    abstract fun resourceTagDao(): ResourceTagDao
    abstract fun appConfigDao(): AppConfigDao

    companion object {
        fun createCallback() = object : Callback() {
            override fun onCreate(db: SupportSQLiteDatabase) {
                super.onCreate(db)
                // 播种内置标签
                db.execSQL("""
                    INSERT OR IGNORE INTO tags (id, name, color, isBuiltIn, createdAt, updatedAt)
                    VALUES ('00000000-0000-0000-0000-000000000001', '收藏', '#FFC107', 1,
                            ${System.currentTimeMillis()}, ${System.currentTimeMillis()})
                """)
            }

            override fun onOpen(db: SupportSQLiteDatabase) {
                super.onOpen(db)
                // 补建内置标签（idempotent）
                db.execSQL("""
                    INSERT OR IGNORE INTO tags (id, name, color, isBuiltIn, createdAt, updatedAt)
                    VALUES ('00000000-0000-0000-0000-000000000001', '收藏', '#FFC107', 1,
                            ${System.currentTimeMillis()}, ${System.currentTimeMillis()})
                """)
                // 启用外键约束
                db.execSQL("PRAGMA foreign_keys = ON;")
            }
        }
    }
}
```

## 8. 数据库初始化

```kotlin
@Module
@InstallIn(SingletonComponent::class)
object DatabaseModule {
    @Provides
    @Singleton
    fun provideDatabase(@ApplicationContext context: Context): AppDatabase {
        return Room.databaseBuilder(context, AppDatabase::class.java, "resource_viewer.db")
            .addCallback(AppDatabase.createCallback())
            .setJournalMode(RoomDatabase.JournalMode.WRITE_AHEAD_LOGGING)
            .addMigrations(MIGRATION_1_2)
            .build()
    }
}
```

## 9. 迁移策略

```kotlin
val MIGRATION_1_2 = object : Migration(1, 2) {
    override fun migrate(db: SupportSQLiteDatabase) {
        // 示例：新增 lastViewedAt 列
        db.execSQL("ALTER TABLE resources ADD COLUMN lastViewedAt INTEGER")
        db.execSQL("CREATE INDEX IF NOT EXISTS resources_last_viewed ON resources (lastViewedAt, id)")
    }
}
```

---

## 10. 注意项

1. **外键约束默认关闭**：Room 默认不启用 `PRAGMA foreign_keys`，需在 `Callback.onOpen` 中手动开启
2. **UUID 为 TEXT**：键集分页时需配合时间戳作为主排序列
3. **密码不入库**：`passwordStored` 仅标记，实际密码在 `EncryptedSharedPreferences` 中
4. **`organizationMode` = null 的含义**：由 `DetectOrganizationModeUseCase` 异步判定
5. **Room 不支持 INTERSECT 语法**：使用 `GROUP BY HAVING COUNT(*) = N` 子查询替代
