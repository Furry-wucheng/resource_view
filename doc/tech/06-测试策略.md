## 1. 概述

定义从 Domain → Repository → ViewModel → Compose → Integration 的五层测试金字塔。

核心原则：**Room 用内存库（不 mock），外部边界用 MockK mock。**

---

## 2. 测试金字塔

```
        ┌──────────┐
        │    E2E   │  极少数：关键用户流程
        ├──────────┤
       ┌┤  Compose  ├┐  少数：核心 UI 组件交互验证
       │└──────────┘│
      ┌┤  ViewModel ├┐  中等：状态转换 + 错误路径 + 筛选逻辑
      │└───────────┘│
     ┌┤  Repo / UC  ├┐  较多：Repository CRUD + UseCase 编排
     │└─────────────┘│
    ┌┤    Domain     ├┐  最广泛：data class、枚举、工具函数
    └┴───────────────┘┘
```

---

## 3. Mock 策略

| 被替换组件 | 方案 | 工具 |
|-----------|------|------|
| Room 数据库 | **内存库** | `Room.inMemoryDatabaseBuilder()` |
| `FileSource` 接口 | mock 接口 | `MockK` |
| `EncryptedSharedPreferences` | mock | `MockK` |
| Repository（在 ViewModel 测试中） | mock | `MockK` |
| ViewModel（在 Compose 测试中） | mock | `MockK` |

### 3.1 选择 MockK

- Kotlin 原生支持，API 简洁
- 支持 mock `final` 类（Room Entity 默认 final）
- `every { mock.method() } returns value` / `verify { mock.method() }`

### 3.2 Room 为什么用内存库

```kotlin
val db = Room.inMemoryDatabaseBuilder(context, AppDatabase::class.java)
    .allowMainThreadQueries()
    .build()
```

- SQLite 内存库极快、确定性强
- 不引入 mock 数据库的维护成本
- 可在 `@Before` 中建表 + 播种，`@After` 中关闭

### 4.2 Repository 测试

```kotlin
@RunWith(AndroidJUnit4::class)
class TagRepositoryTest {
    private lateinit var db: AppDatabase
    private lateinit var repo: TagRepository

    @Before
    fun setup() {
        db = Room.inMemoryDatabaseBuilder(
            ApplicationProvider.getApplicationContext(),
            AppDatabase::class.java
        ).allowMainThreadQueries().build()

        // 触发 onCreate（建表 + 内置标签播种）
        db.openHelper.writableDatabase
        repo = TagRepository(db.tagDao(), db.resourceTagDao())
    }

    @After
    fun teardown() { db.close() }

    @Test
    fun `filterByTags with empty list should return all resources`() = runTest {
        createTestResource(db, name = "资源A")
        createTestResource(db, name = "资源B")

        repo.filterByTags(emptyList()).first().let { result ->
            assertEquals(2, result.size)
        }
    }

    @Test
    fun `filterByTags with two tags should return intersection`() = runTest {
        val tagA = repo.createTag("热血", "#FF0000")
        val tagB = repo.createTag("已完结", "#0000FF")
        val res1 = createTestResource(db, name = "海贼王")
        val res2 = createTestResource(db, name = "火影")

        repo.addResourceTag(res1, tagA)
        repo.addResourceTag(res1, tagB)
        repo.addResourceTag(res2, tagA)

        repo.filterByTags(listOf(tagA, tagB)).first().let { result ->
            assertEquals(1, result.size)
            assertEquals(res1, result.first().id)
        }
    }

    @Test
    fun `deleteSource should cascade delete resources and resourceTags`() = runTest {
        val sourceId = createTestSource(db)
        val resourceId = createTestResource(db, sourceId = sourceId)
        val tagId = repo.createTag("标签", "#FF0000")
        repo.addResourceTag(resourceId, tagId)

        repo.deleteSource(sourceId)

        assertNull(repo.getResource(resourceId))
        assertEquals(0, repo.getTagCount(tagId))
    }
}
```

class DetectOrganizationModeUseCaseTest {
    private lateinit var mockSource: FileSource
    private lateinit var useCase: DetectOrganizationModeUseCase

    @Before
    fun setup() {
        mockSource = mockk()
        useCase = DetectOrganizationModeUseCase()
    }

    @Test
    fun `folder with only images should be flat grid`() = runTest {
        every { mockSource.listDirectory("/test") } returns listOf(
            FileEntry("001.jpg", isDirectory = false, size = 1000, modified = System.currentTimeMillis()),
            FileEntry("002.jpg", isDirectory = false, size = 1000, modified = System.currentTimeMillis()),
        )

        val mode = useCase.detect("/test", mockSource)
        assertEquals(OrganizationMode.FLATGRID, mode)
    }

    @Test
    fun `folder with subfolders containing images should be chapter`() = runTest {
        every { mockSource.listDirectory("/test") } returns listOf(
            FileEntry("卷01", isDirectory = true, size = 0, modified = System.currentTimeMillis()),
            FileEntry("卷02", isDirectory = true, size = 0, modified = System.currentTimeMillis()),
        )
        every { mockSource.listDirectory("/test/卷01") } returns listOf(
            FileEntry("001.jpg", isDirectory = false, size = 1000, modified = System.currentTimeMillis()),
        )

        val mode = useCase.detect("/test", mockSource)
        assertEquals(OrganizationMode.CHAPTER, mode)
    }

    @Test
    fun `mixed content should be flat grid`() = runTest {
        every { mockSource.listDirectory("/test") } returns listOf(
            FileEntry("设定集.pdf", isDirectory = false, size = 5000, modified = System.currentTimeMillis()),
            FileEntry("画集", isDirectory = true, size = 0, modified = System.currentTimeMillis()),
            FileEntry("花絮.mp4", isDirectory = false, size = 10000, modified = System.currentTimeMillis()),
        )

        val mode = useCase.detect("/test", mockSource)
        assertEquals(OrganizationMode.FLATGRID, mode)
    }
}

### 4.4 ViewModel 测试

```kotlin
class HomeViewModelTest {
    private lateinit var mockResourceRepo: ResourceRepository
    private lateinit var mockTagRepo: TagRepository
    private lateinit var mockFilterUseCase: FilterResourcesByTagsUseCase
    private lateinit var viewModel: HomeViewModel

    @Before
    fun setup() {
        mockResourceRepo = mockk()
        mockTagRepo = mockk()
        mockFilterUseCase = mockk()
        viewModel = HomeViewModel(mockResourceRepo, mockTagRepo, mockFilterUseCase)
    }

    @Test
    fun `loadResources should emit loading then success state`() = runTest {
        coEvery { mockResourceRepo.getAllResources() } returns Result.Ok(listOf(testResource()))

        val states = mutableListOf<UiState>()
        viewModel.uiStateFlow.collect { states.add(it) }

        viewModel.loadResources()

        assertTrue(states.contains(UiState.LOADING))
        assertTrue(states.contains(UiState.SUCCESS))
        assertEquals(1, viewModel.resources.size)
    }

    @Test
    fun `SMB failure should emit error state with retry enabled`() = runTest {
        coEvery { mockResourceRepo.getAllResources() } returns
            Result.Err(DomainError.SourceUnreachableError("SMB 不可达"))

        viewModel.loadResources()

        assertEquals(UiState.ERROR, viewModel.uiState)
        assertTrue(viewModel.canRetry)
        assertTrue(viewModel.errorMessage!!.contains("不可达"))
    }

    @Test
    fun `selectTag should update selectedTagIds and re-filter`() = runTest {
        coEvery { mockFilterUseCase.execute(listOf("tag1")) } returns
            Result.Ok(listOf(testResource()))

        viewModel.selectTag("tag1")

        assertEquals(setOf("tag1"), viewModel.selectedTagIds)
        assertEquals(1, viewModel.resources.size)
    }
}
```

## 8. 注意项

1. **Room 不 mock** — `Room.inMemoryDatabaseBuilder()` 替代，`@Before` 建表，`@After` 关闭
2. **ViewModel 测试用 `runTest`** — 协程测试标准方式
3. **Compose 测试用 `createComposeRule`** — 验证 UI 渲染和交互
4. **mock 类使用 MockK** — `mockk<T>()` / `every { } returns` / `verify { }`
5. **错误路径测试与正常路径同等重要** — 每个 ViewModel 的 `load → error → retry` 回路必须测
